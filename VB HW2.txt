Sub stocks()

    Dim WS_Count As Integer
    Dim I As Integer

    ' Keep track of the location for each ticker volume in the summary table
  Dim Summary_Table_Row As Integer
  Summary_Table_Row = 2

    ' Set WS_Count equal to the number of worksheets in the active
     ' workbook.
    WS_Count = ActiveWorkbook.Worksheets.Count

     For I = 1 To WS_Count
     ' declare variables to find last row in a worksheet
        Dim LastRow As Long
        Dim volume As Range
        dim ticker as Long

        'set initial variable for holding stock volume
        Dim total_stock_volume as Long
        total_stock_volume = 0
        
        ' setting the worksheet because i can
        Set volume = Range("A2")

        'Find Last Row in Column A
         LastRow = Cells(Rows.Count, volume.Column).End(xlUp).Row

        'Select Range
        Range(volume, Cells(LastRow, 1)).Select
            
            'start the for reals code to do the reals things
            For j = 2 To LastRow
            
            'check if the current cell does NOT match cell below
            If Cells(j + 1, 1).Value <> Cells(j, 1).Value Then

                ' Set the ticker name
                ticker = Cells(j, 1).Value

                ' Add to the Brand Total
                total_stock_volume = total_stock_volume + Cells(j, 7).Value

                ' Print the Credit Card Brand in the Summary Table
                Range("I" & Summary_Table_Row).Value = ticker

                ' Print the Brand Amount to the Summary Table
                Range("J" & Summary_Table_Row).Value = total_stock_volume

                ' Add one to the summary table row
                Summary_Table_Row = Summary_Table_Row + 1
      
                ' Reset the Brand Total
                total_stock_volume = 0

            ' If the cell immediately following a row is the same brand...
            Else

                'Add to the Brand Total
                total_stock_volume = total_stock_volume + Cells(j, 7).Value

            
            End If
            
            Next j

         Next I
     
      End Sub
